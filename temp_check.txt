let cardIdx = 0;

function rankLabel(r){ return r==='S'?'⭐⭐⭐ S级':r==='A'?'⭐⭐ A级':'⭐ B级'; }
function rankBg(r){ return r==='S'?'rgba(248,113,113,.15)':r==='A'?'rgba(200,169,110,.15)':'rgba(100,116,139,.15)'; }
function rankColor(r){ return r==='S'?'#f87171':r==='A'?'#c8a96e':'#94a3b8'; }

function ringLabel(r){
  if(r==='hist') return '<span class="ring-tag ring-hist">🟠 历史验证</span>';
  if(r==='biz')  return '<span class="ring-tag ring-biz">🟢 业务交付</span>';
  if(r==='ref')  return '<span class="ring-tag ring-ref">🟣 对标账号</span>';
  return '';
}

function crossBadge(cr){
  if(cr==='SS') return '<span class="cross-badge">🏆 双S·必拍</span>';
  return '';
}

function toggle(i){
  document.getElementById('pc'+i).classList.toggle('open');
}

function goTo(id, btn){
  document.getElementById(id).scrollIntoView({behavior:'smooth', block:'start'});
  document.querySelectorAll('.nav-btn').forEach(b=>b.classList.remove('active'));
  btn.classList.add('active');
}

const sections = ['g1','g2','g3','g4','g5','method'];
const navBtns = document.querySelectorAll('.nav-btn');
window.addEventListener('scroll',()=>{
  let cur = 0;
  sections.forEach((id,i)=>{
    const el = document.getElementById(id);
    if(el && window.scrollY >= el.offsetTop - 120) cur = i;
  });
  navBtns.forEach((b,i)=>b.classList.toggle('active', i===cur));
},{passive:true});
</script>
